56 inline cl_platform_id select_platform() {
57 cl_uint n_platforms;
58 check(clGetPlatformIDs(0, nullptr, &n_platforms), "Getting platform count");
59 std::vector<cl_platform_id> platforms(n_platforms);
60 check(clGetPlatformIDs(n_platforms, platforms.data(), nullptr), "Getting platforms");
61
62 const char* env = std::getenv("OCL_PLATFORM");
63 cl_uint index = (env && env[0] != '\0') ? std::atoi(env) : 0;

65 if (index >= n_platforms) {
66 std::cerr << "Invalid platform index: " << index << std::endl;
67 std::exit(EXIT_FAILURE);
68 }
69
70 char name[BUFSIZE];
71 check(clGetPlatformInfo(platforms[index], CL_PLATFORM_NAME, BUFSIZE, name, nullptr), "Getting platform name");
72 std::cout << "Selected platform " << index << ": " << name << std::endl;

139 inline cl_program create_program(const std::string& filename, cl_context context, cl_device_id device) {
140 std::ifstream file(filename);

143 std::exit(EXIT_FAILURE);
144 }

146 std::ostringstream oss;

174 inline cl_ulong runtime_ns(cl_event evt) {
175 cl_ulong start = 0, end = 0;
176 check(clGetEventProfilingInfo(evt, CL_PROFILING_COMMAND_START, sizeof(start), &start, nullptr), "Profiling start");
177 check(clGetEventProfilingInfo(evt, CL_PROFILING_COMMAND_END, sizeof(end), &end, nullptr), "Profiling end");
178 return end - start;
179 }
